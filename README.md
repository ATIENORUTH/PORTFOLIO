function displayGreeting() {
    const now = new Date();
    const hours = now.getHours();
    let greeting = 'Hello!';

    if (hours < 12) {
        greeting = 'Good Morning!';
    } else if (hours < 18) {
        greeting = 'Good Afternoon!';
    } else {
        greeting = 'Good Evening!';
    }

    alert(`${greeting} Welcome to my portfolio!`);
}

window.addEventListener('load', displayGreeting);
